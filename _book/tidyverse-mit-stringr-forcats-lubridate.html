zum Subsetten

x <- c("apple", "banana", "pears")
str_sub(x,start= 1,end= 3)
## [1] "app" "ban" "pea"
#überschreiben der 2- 4 buchstaben als Großbuchstaben
str_sub(x,start= 2,end= 4) <- str_to_upper(str_sub(x,2,4)) 
x
## [1] "aPPLe"  "bANAna" "pEARs"
